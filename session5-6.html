Chapter 1: Modules

1.1 – What is a Module?

A module is a Python file (.py) that contains functions, variables, and classes you can reuse in other programs.

# mytools.py  <- a module you write
def add(a, b):
    return a + b

def multiply(a, b):
    return a * b

PI = 3.14159
# main.py  <- your program
import mytools

print(mytools.add(3, 4))       # 7
print(mytools.multiply(3, 4))  # 12
print(mytools.PI)              # 3.14159

Note

Modules help you organize code and avoid rewriting the same logic in every project.

1.2 – Import Syntax

Four ways to import:

# 1. Import the whole module -- access with dot notation
import math
print(math.sqrt(16))    # 4.0
print(math.pi)          # 3.141592653589793

# 2. Import specific names -- use them directly (no prefix)
from math import sqrt, pi
print(sqrt(25))         # 5.0
print(pi)               # 3.141592653589793

# 3. Rename with alias -- shorter to type
import math as m
print(m.floor(3.9))     # 3

# 4. Import everything (not recommended)
from math import *
print(ceil(2.1))        # 3

Important

Prefer import math or from math import sqrt over import * – it keeps your code clear about where names come from.

1.3 – Standard Library: math & random

math – mathematical functions:

import math

math.sqrt(49)        # 7.0
math.ceil(4.2)       # 5   -- round up
math.floor(4.8)      # 4   -- round down
math.pow(2, 8)       # 256.0
math.pi              # 3.14159...
math.log(100, 10)    # 2.0
abs(-7)              # 7   -- built-in, no import needed

random – random number generation:

import random

random.randint(1, 6)          # random int 1-6 (inclusive)
random.random()               # float in [0.0, 1.0)
random.choice(["a","b","c"])  # pick one item randomly
random.shuffle([1, 2, 3, 4])  # shuffle list in-place
random.uniform(1.0, 5.0)      # random float between 1.0 and 5.0

1.4 – Standard Library: os & datetime

os – interact with the operating system:

import os

os.getcwd()                          # current working directory
os.listdir(".")                      # list files in current folder
os.path.exists("data.txt")           # True if file/folder exists
os.path.join("folder", "file.txt")  # build a safe file path
os.mkdir("new_folder")              # create a new directory

datetime – work with dates and times:

from datetime import datetime, date

today = date.today()
print(today)                      # 2026-05-22

now = datetime.now()
print(now.year)                   # 2026
print(now.strftime("%d/%m/%Y"))   # 22/05/2026
print(now.strftime("%H:%M:%S"))   # 14:30:00

1.5 – Creating Your Own Module

Step 1 – Create grader.py:

# grader.py

def get_grade(score):
    if score >= 90: return "A"
    elif score >= 80: return "B"
    elif score >= 70: return "C"
    elif score >= 60: return "D"
    else: return "F"

def is_passing(score):
    return score >= 60

def summary(name, score):
    grade = get_grade(score)
    status = "Passing" if is_passing(score) else "Failing"
    return f"{name} | {score} | Grade {grade} | {status}"

Step 2 – Use it in main.py:

import grader

print(grader.get_grade(85))            # B
print(grader.is_passing(45))           # False
print(grader.summary("Alice", 92))     # Alice | 92 | Grade A | Passing

1.6 – The __name__ Guard

When Python runs a file, it sets a special variable __name__:

  • File is run directly –> __name__ == "__main__"
  • File is imported –> __name__ == "grader" (the module name)
# grader.py

def get_grade(score):
    if score >= 90: return "A"
    elif score >= 80: return "B"
    else: return "F"

if __name__ == "__main__":
    # This block runs ONLY when grader.py is run directly
    # It is SKIPPED when grader is imported
    print(get_grade(95))    # A
    print(get_grade(60))    # F

Note

This pattern lets you test a module file without running that test code every time it is imported by another file.

Chapter 2: File Handling

2.1 – Why File Handling?

Programs that only use variables lose all data when they stop running. Files give programs persistent memory.

Use Case Example
Save data permanently Store student scores between sessions
Read configuration Load settings at startup
Exchange data Export results to CSV or JSON
Keep a log Record events or errors over time 
# Without files -- data is lost when program ends
scores = [85, 92, 78]

# With files -- data persists after the program exits
with open("scores.txt", "w") as f:
    for s in scores:
        f.write(str(s) + "\n")

2.2 – The open() Function & File Modes

file = open(filename, mode)
Mode Meaning
"r" Read only (default) – file must exist
"w" Write – creates file; overwrites if it already exists
"a" Append – adds to end; creates file if it doesn’t exist
"r+" Read and write – file must exist
"x" Create exclusively – fails if file already exists

Important

Always close the file after use: file.close() – or better, use the with statement which closes automatically.

2.3 – Reading Files

# read() -- entire file as one string
with open("students.txt", "r") as f:
    content = f.read()
    print(content)

# readline() -- one line at a time (includes \n)
with open("students.txt", "r") as f:
    line1 = f.readline()
    line2 = f.readline()

# readlines() -- all lines as a list
with open("students.txt", "r") as f:
    lines = f.readlines()
    print(lines)   # ['Alice\n', 'Bob\n', 'Charlie\n']

# Loop directly -- most memory-efficient for large files
with open("students.txt", "r") as f:
    for line in f:
        print(line.strip())   # strip() removes the \n

2.4 – Writing Files

# write() -- write a string to a file
with open("output.txt", "w") as f:
    f.write("Hello, World!\n")
    f.write("Second line.\n")

# writelines() -- write a list of strings
lines = ["Alice\n", "Bob\n", "Charlie\n"]
with open("names.txt", "w") as f:
    f.writelines(lines)

# Append mode -- add content without overwriting
with open("log.txt", "a") as f:
    f.write("Session started.\n")

Note

write() does not add a newline automatically – you must include \n yourself in the string.

2.5 – The with Statement

The with statement is a context manager – it guarantees the file is closed even if an error occurs inside the block:

# Without with -- risky: file may stay open if an error occurs
f = open("data.txt", "r")
content = f.read()
f.close()   # <- skipped if read() raises an exception

# With with -- safe and clean: always closes
with open("data.txt", "r") as f:
    content = f.read()
# file is closed here automatically, no matter what

Tip

Always use with open(...) – it is safer and more readable than calling .close() manually.

2.6 – Reading & Writing Records

Write a list of records:

students = [("Alice", 88), ("Bob", 75), ("Charlie", 92)]

with open("results.txt", "w") as f:
    for name, score in students:
        f.write(f"{name},{score}\n")

Read back and parse each line:

with open("results.txt", "r") as f:
    for line in f:
        parts = line.strip().split(",")
        name  = parts[0]
        score = int(parts[1])
        print(f"{name}: {score}")

Check if a file exists before opening:

import os

if os.path.exists("results.txt"):
    with open("results.txt", "r") as f:
        print(f.read())
else:
    print("File not found.")

Chapter 3: Error Handling

3.1 – What is an Exception?

An exception is a runtime error that stops your program immediately:

print(10 / 0)       # ZeroDivisionError
int("hello")        # ValueError
lst = [1, 2, 3]
print(lst[10])      # IndexError
d = {"a": 1}
print(d["z"])       # KeyError
open("missing.txt") # FileNotFoundError

Without error handling the program crashes. With error handling you can respond gracefully and keep the program running:

try:
    result = 10 / 0
except ZeroDivisionError:
    print("Cannot divide by zero!")
    result = 0

3.2 – try / except

Catch a specific exception:

try:
    x = int(input("Enter a number: "))
    print(10 / x)
except ValueError:
    print("That is not a valid number.")
except ZeroDivisionError:
    print("Cannot divide by zero.")

Catch any exception and inspect the message:

try:
    result = some_function()
except Exception as e:
    print(f"Something went wrong: {e}")

Tip

Catch specific exception types when possible – catching Exception too broadly can hide real bugs.

3.3 – Common Built-in Exceptions

Exception When it occurs
ValueError Right type but wrong value: int("abc")
TypeError Wrong type in operation: "5" + 5
ZeroDivisionError Division by zero: 1 / 0
IndexError List index out of range: lst[99]
KeyError Missing dictionary key: d["missing"]
FileNotFoundError File does not exist: open("x.txt")
NameError Variable not defined: print(xyz)
AttributeError Object has no such method: (5).upper()

3.4 – else and finally

try:
    x = int(input("Enter a number: "))
    result = 100 / x
except ValueError:
    print("Not a valid number.")
except ZeroDivisionError:
    print("Cannot divide by zero.")
else:
    # Runs ONLY if the try block raised NO exception
    print(f"Result: {result}")
finally:
    # ALWAYS runs -- whether an exception occurred or not
    print("Calculation attempt finished.")

Note

finally is ideal for cleanup work: closing a database connection, printing a status line, or releasing a resource – it always runs.

3.5 – Raising Exceptions

Use raise to signal an error deliberately when input breaks your rules:

def set_age(age):
    if not isinstance(age, int):
        raise TypeError("Age must be an integer.")
    if age < 0 or age > 150:
        raise ValueError(f"Age {age} is not a realistic value.")
    return age

try:
    set_age("twenty")
except TypeError as e:
    print(f"Type error: {e}")

try:
    set_age(-5)
except ValueError as e:
    print(f"Value error: {e}")

Tip

Using raise makes your functions fail early and clearly – much easier to debug than a silent wrong value.

3.6 – Error Handling with Files

Combining file handling and error handling together:

def read_scores(filename):
    try:
        with open(filename, "r") as f:
            scores = []
            for line in f:
                scores.append(int(line.strip()))
            return scores
    except FileNotFoundError:
        print(f"Error: '{filename}' does not exist.")
        return []
    except ValueError:
        print("Error: file contains non-numeric data.")
        return []

scores = read_scores("scores.txt")
if scores:
    avg = sum(scores) / len(scores)
    print(f"Average: {avg:.2f}")

Exercise 1

Build a grader Module

Task: Create a module file grader.py with three functions:

  1. get_grade(score) – returns "A", "B", "C", "D", or "F" based on 90/80/70/60 thresholds
  2. is_passing(score) – returns True if score >= 60
  3. summary(name, score) – returns a formatted string: "Alice | 92 | Grade A | Passing"

Expected output:

Both files must be in the same folder. Use import grader then call grader.get_grade(...).

def get_grade(score):

Exercise 2

Write & Read Student Records

Task: Write a program that:

  1. Saves a list of student records to "students.txt" – one record per line in the format name,score
  2. Reads the file back line by line
  3. Prints each student’s name, score, and grade (using the grader module from Exercise 1)
  4. Prints the class average at the end

Expected output:

Alice: 92 -> Grade A
Bob: 75 -> Grade C
Charlie: 83 -> Grade B
Diana: 58 -> Grade F

Class average: 77.00

Hint

Use line.strip().split(",") to parse each line. Convert the score to int() after splitting.

Exercise 2 – Sample Solution

import grader

students = [("Alice", 92), ("Bob", 75), ("Charlie", 83), ("Diana", 58)]

# Write to file
with open("students.txt", "w") as f:
    for name, score in students:
        f.write(f"{name},{score}\n")

# Read and display
total = 0
count = 0
with open("students.txt", "r") as f:
    for line in f:
        name, score_str = line.strip().split(",")
        score = int(score_str)
        grade = grader.get_grade(score)
        print(f"{name}: {score} -> Grade {grade}")
        total += score
        count += 1

print(f"\nClass average: {total / count:.2f}")

Exercise 3

Safe Score Reader with Error Handling

Task: Write a function load_scores(filename) that:

  1. Tries to open and read filename
  2. Parses each line as an integer (one score per line)
  3. Handles FileNotFoundError – prints a message and returns []
  4. Handles ValueError – prints a message and skips the bad line
  5. Returns a list of valid scores

Then call it with a real file and with a missing filename. Print the average if scores are found.

Expected output (if file has scores 85, 92, 70, bad_line, 78):

Warning: could not read line "bad_line" as a number. Skipping.
Scores loaded: [85, 92, 70, 78]
Average: 81.25

Exercise 3 – Sample Solution

def load_scores(filename):
    try:
        scores = []
        with open(filename, "r") as f:
            for line in f:
                line = line.strip()
                try:
                    scores.append(int(line))
                except ValueError:
                    print(f'Warning: could not read line "{line}" as a number. Skipping.')
        return scores
    except FileNotFoundError:
        print(f"Error: '{filename}' does not exist.")
        return []

# Test with existing file
scores = load_scores("scores.txt")
if scores:
    print(f"Scores loaded: {scores}")
    print(f"Average: {sum(scores) / len(scores):.2f}")

# Test with missing file
load_scores("missing.txt")

Skills practiced: nested try/except, FileNotFoundError, ValueError, defensive file reading

Exercise 4

Append-Only Event Log

Task: Write two functions:

  1. log_event(filename, message) – appends a timestamped line to a text log file using datetime. Format: "[2026-05-22 14:30:00] message". Handles any OSError gracefully.
  2. show_log(filename) – reads and prints every line in the log file. Handles FileNotFoundError.

Call log_event three times with different messages, then call show_log.

Expected output:

[2026-05-22 14:30:00] Program started.
[2026-05-22 14:30:01] Alice scored 92.
[2026-05-22 14:30:01] Report complete.

Hint

Use from datetime import datetime and datetime.now().strftime("%Y-%m-%d %H:%M:%S"). Append mode "a" creates the file if it doesn’t exist yet.

Exercise 4 – Sample Solution

from datetime import datetime

def log_event(filename, message):
    try:
        timestamp = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
        with open(filename, "a") as f:
            f.write(f"[{timestamp}] {message}\n")
    except OSError as e:
        print(f"Could not write to log: {e}")

def show_log(filename):
    try:
        with open(filename, "r") as f:
            for line in f:
                print(line.strip())
    except FileNotFoundError:
        print(f"Log file '{filename}' not found.")

log_event("events.log", "Program started.")
log_event("events.log", "Alice scored 92.")
log_event("events.log", "Report complete.")
show_log("events.log")

Skills practiced: datetime, append mode, OSError, FileNotFoundError, timestamped logging

Exercise 5

Mini Grade Report System

Task: Combine everything into one complete program using only files and the grader module:

  1. save_report(students, filename) – writes a formatted text report to a file, one student per line: "Alice | 92 | Grade A | Passing"
  2. load_and_print_report(filename) – reads and prints the file; handles FileNotFoundError
  3. get_stats(filename) – reads the file, parses scores from each line, then prints average, top student, and count of passing students; handles missing file and bad lines

Run the full cycle: build the list, save, load and print, show stats.

Exercise 5 – Sample Solution

import grader

def save_report(students, filename):
    with open(filename, "w") as f:
        for name, score in students:
            grade  = grader.get_grade(score)
            status = "Passing" if grader.is_passing(score) else "Failing"
            f.write(f"{name:<10}|{score}|{grade}|{status}\n")
    print(f"Report saved to {filename}")

def load_and_print_report(filename):
    try:
        print(f"\n{'Name':<10} {'Score':>5} {'Grade':>6} {'Status'}")
        print("-" * 35)
        with open(filename, "r") as f:
            for line in f:
                name, score, grade, status = line.strip().split("|")
                print(f"{name:<10} {score:>5} {grade:>6} {status}")
    except FileNotFoundError:
        print(f"'{filename}' not found.")

def get_stats(filename):
    try:
        scores = []
        names  = []
        with open(filename, "r") as f:
            for line in f:
                parts = line.strip().split("|")
                names.append(parts[0].strip())
                scores.append(int(parts[1]))
        avg     = sum(scores) / len(scores)
        top_idx = scores.index(max(scores))
        passing = sum(1 for s in scores if s >= 60)
        print(f"\nAverage: {avg:.1f} | Top: {names[top_idx]} | Passing: {passing}/{len(scores)}")
    except FileNotFoundError:
        print("No report file found.")

students = [("Alice",92),("Bob",55),("Charlie",83),("Diana",70),("Eve",48)]
save_report(students, "report.txt")
load_and_print_report("report.txt")
get_stats("report.txt")

Skills practiced: all topics combined – custom module, text file I/O, parsing, error handling, stats
